```sql
-- Generate unique referral code
CREATE OR REPLACE FUNCTION generate_referral_code()
RETURNS TEXT AS $$
DECLARE
  chars TEXT := 'ABCDEFGHJKLMNPQRSTUVWXYZ23456789';
  result TEXT := '';
  i INTEGER;
BEGIN
  FOR i IN 1..8 LOOP
    result := result || substr(chars, floor(random() * length(chars) + 1)::integer, 1);
  END LOOP;
  RETURN result;
END;
$$ LANGUAGE plpgsql;

-- Assign position on signup
CREATE OR REPLACE FUNCTION assign_waitlist_position()
RETURNS TRIGGER AS $$
BEGIN
  NEW.position := (SELECT COALESCE(MAX(position), 0) + 1 FROM public.waitlist_subscribers);
  IF NEW.referral_code IS NULL THEN
    NEW.referral_code := generate_referral_code();
  END IF;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER waitlist_assign_position
  BEFORE INSERT ON public.waitlist_subscribers
  FOR EACH ROW
  EXECUTE FUNCTION assign_waitlist_position();

-- Update referral count when someone signs up with referral
CREATE OR REPLACE FUNCTION update_referral_count()
RETURNS TRIGGER AS $$
BEGIN
  IF NEW.referred_by IS NOT NULL THEN
    UPDATE public.waitlist_subscribers
    SET 
      referral_count = referral_count + 1,
      priority_score = priority_score + 10,
      updated_at = now()
    WHERE id = NEW.referred_by;
  END IF;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER waitlist_update_referral
  AFTER INSERT ON public.waitlist_subscribers
  FOR EACH ROW
  EXECUTE FUNCTION update_referral_count();

-- Get waitlist stats for public counter
CREATE OR REPLACE FUNCTION get_waitlist_count()
RETURNS INTEGER AS $$
BEGIN
  RETURN (SELECT COUNT(*) FROM public.waitlist_subscribers WHERE status != 'unsubscribed');
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;
```

## Edge Functions

### waitlist-signup
Handles new waitlist signups from public page.

```typescript
// POST /functions/v1/waitlist-signup
import { serve } from 'https://deno.land/std@0.168.0/http/server.ts';
import { createClient } from 'https://esm.sh/@supabase/supabase-js@2';

const corsHeaders = {
  'Access-Control-Allow-Origin': '*',
  'Access-Control-Allow-Headers': 'authorization, x-client-info, apikey, content-type',
};

serve(async (req) => {
  if (req.method === 'OPTIONS') {
    return new Response('ok', { headers: corsHeaders });
  }

  try {
    const supabase = createClient(
      Deno.env.get('SUPABASE_URL')!,
      Deno.env.get('SUPABASE_SERVICE_ROLE_KEY')!
    );

    const { email, name, referral_code, utm_source, utm_medium, utm_campaign } = await req.json();

    // Check if engine is running
    const { data: settings } = await supabase
      .from('waitlist_settings')
      .select('*')
      .single();

    if (!settings?.is_running) {
      return new Response(
        JSON.stringify({ error: 'Waitlist is not currently accepting signups' }),
        { status: 400, headers: { ...corsHeaders, 'Content-Type': 'application/json' } }
      );
    }

    // Check if already registered
    const { data: existing } = await supabase
      .from('waitlist_subscribers')
      .select('id, position, referral_code, referral_count')
      .eq('email', email.toLowerCase())
      .single();

    if (existing) {
      return new Response(
        JSON.stringify({
          success: true,
          existing: true,
          position: existing.position,
          referral_code: existing.referral_code,
          referral_count: existing.referral_count,
        }),
        { headers: { ...corsHeaders, 'Content-Type': 'application/json' } }
      );
    }

    // Find referrer if code provided
    let referrerId = null;
    if (referral_code) {
      const { data: referrer } = await supabase
        .from('waitlist_subscribers')
        .select('id')
        .eq('referral_code', referral_code.toUpperCase())
        .single();
      
      if (referrer) {
        referrerId = referrer.id;
      }
    }

    // Create subscriber
    const { data: subscriber, error: insertError } = await supabase
      .from('waitlist_subscribers')
      .insert({
        email: email.toLowerCase(),
        name,
        referred_by: referrerId,
        utm_source,
        utm_medium,
        utm_campaign,
        ip_address: req.headers.get('x-forwarded-for') || req.headers.get('x-real-ip'),
        user_agent: req.headers.get('user-agent'),
      })
      .select()
      .single();

    if (insertError) throw insertError;

    // Update daily stats
    await supabase.rpc('increment_waitlist_stats', {
      is_referral: referrerId !== null,
    });

    // Trigger welcome email
    if (settings.welcome_email_enabled) {
      await supabase.functions.invoke('waitlist-send-email', {
        body: { subscriber_id: subscriber.id, email_type: 'welcome' },
      });
    }

    // Trigger Slack notification
    if (settings.slack_enabled) {
      await supabase.functions.invoke('waitlist-slack-notify', {
        body: { subscriber_id: subscriber.id },
      });
    }

    // Check referrer rewards
    if (referrerId) {
      await supabase.functions.invoke('waitlist-check-rewards', {
        body: { subscriber_id: referrerId },
      });
    }

    return new Response(
      JSON.stringify({
        success: true,
        position: subscriber.position,
        referral_code: subscriber.referral_code,
        referral_count: 0,
      }),
      { headers: { ...corsHeaders, 'Content-Type': 'application/json' } }
    );

  } catch (error) {
    // Log error
    const supabase = createClient(
      Deno.env.get('SUPABASE_URL')!,
      Deno.env.get('SUPABASE_SERVICE_ROLE_KEY')!
    );

    await supabase.from('waitlist_errors').insert({
      function_name: 'waitlist-signup',
      error_message: error.message,
      error_details: { stack: error.stack },
    });

    return new Response(
      JSON.stringify({ error: 'Failed to join waitlist' }),
      { status: 500, headers: { ...corsHeaders, 'Content-Type': 'application/json' } }
    );
  }
});
```

### waitlist-send-email
Sends emails via Resend API.

```typescript
// POST /functions/v1/waitlist-send-email
import { serve } from 'https://deno.land/std@0.168.0/http/server.ts';
import { createClient } from 'https://esm.sh/@supabase/supabase-js@2';

serve(async (req) => {
  try {
    const supabase = createClient(
      Deno.env.get('SUPABASE_URL')!,
      Deno.env.get('SUPABASE_SERVICE_ROLE_KEY')!
    );

    const { subscriber_id, email_type } = await req.json();

    // Get subscriber
    const { data: subscriber } = await supabase
      .from('waitlist_subscribers')
      .select('*')
      .eq('id', subscriber_id)
      .single();

    if (!subscriber) throw new Error('Subscriber not found');

    // Get settings
    const { data: settings } = await supabase
      .from('waitlist_settings')
      .select('*')
      .single();

    // Get Resend API key from secrets
    const resendApiKey = Deno.env.get('RESEND_API_KEY');
    if (!resendApiKey) throw new Error('RESEND_API_KEY not configured');

    let subject: string;
    let body: string;

    if (email_type === 'welcome') {
      subject = settings.welcome_email_subject;
      body = settings.welcome_email_body;
    } else if (email_type === 'followup') {
      subject = settings.followup_subject;
      body = settings.followup_body;
    } else {
      throw new Error('Invalid email type');
    }

    // Personalize email
    const referralLink = `${Deno.env.get('SUPABASE_URL')?.replace('.supabase.co', '')}/waitlist?ref=${subscriber.referral_code}`;
    
    body = body
      .replace('{{name}}', subscriber.name || 'there')
      .replace('{{position}}', subscriber.position.toString())
      .replace('{{referral_link}}', referralLink)
      .replace('{{referral_code}}', subscriber.referral_code);

    // Add backlink to email footer
    const emailHtml = `
      <div style="font-family: sans-serif; max-width: 600px; margin: 0 auto;">
        <div style="padding: 20px;">
          ${body.split('\n').map(p => `<p>${p}</p>`).join('')}
          
          <div style="margin-top: 30px; padding: 20px; background: #f5f5f5; border-radius: 8px;">
            <p style="margin: 0 0 10px; font-weight: bold;">Share and move up the list:</p>
            <p style="margin: 0; word-break: break-all;">
              <a href="${referralLink}">${referralLink}</a>
            </p>
          </div>
        </div>
        
        <div style="padding: 20px; text-align: center; color: #999; font-size: 12px;">
          <p>Powered by <a href="https://lazyunicorn.ai" style="color: #999;">LazyUnicorn.ai</a></p>
        </div>
      </div>
    `;

    // Send via Resend
    const response = await fetch('https://api.resend.com/emails', {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${resendApiKey}`,
        'Content-Type': 'application/json',
      },
      body: JSON.stringify({
        from: 'waitlist@resend.dev', // Update with verified domain
        to: subscriber.email,
        subject: subject,
        html: emailHtml,
      }),
    });

    if (!response.ok) {
      const error = await response.json();
      throw new Error(`Resend API error: ${JSON.stringify(error)}`);
    }

    // Update subscriber record
    const updateField = email_type === 'welcome' 
      ? { welcome_email_sent: true, welcome_email_sent_at: new Date().toISOString() }
      : { followup_email_sent: true, followup_email_sent_at: new Date().toISOString() };

    await supabase
      .from('waitlist_subscribers')
      .update(updateField)
      .eq('id', subscriber_id);

    // Update daily stats
    await supabase.rpc('increment_emails_sent');

    return new Response(JSON.stringify({ success: true }), {
      headers: { 'Content-Type': 'application/json' },
    });

  } catch (error) {
    const supabase = createClient(
      Deno.env.get('SUPABASE_URL')!,
      Deno.env.get('SUPABASE_SERVICE_ROLE_KEY')!
    );

    await supabase.from('waitlist_errors').insert({
      function_name: 'waitlist-send-email',
      error_message: error.message,
      error_details: { stack: error.stack },
    });

    return new Response(
      JSON.stringify({ error: error.message }),
      { status: 500, headers: { 'Content-Type': 'application/json' } }
    );
  }
});
```
